    This grid was created using these simple handlebars templates:

    {{#grid.container}}
<div class="bs-docs-grid">{{#rows}}
  <div class="row show-grid">{{#columns}}
    <div class="span"> {{content}} </div>
    {{/columns}}</div>
  {{/rows}}</div>
{{/grid.container}}
